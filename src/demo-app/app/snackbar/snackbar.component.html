  export class SnackbarDemo {

    constructor(private mdlSnackbarService: MdlSnackbarService, private vcRef: ViewContainerRef) {
      mdlSnackbarService.setDefaultViewContainerRef(vcRef);
    }

    showSnackbar() {
      this.mdlSnackbarService.showSnackbar({
        message:'The Message',
        action:{
          handler:()=>{
            this.mdlSnackbarService.showToast('You hit the ok Button');
          },
          text: 'OK'
        }
      });
    }
  }

Method Description
setDefaultViewContainerRef(
vcRef: ViewContainerRef)		 Every SnackBar needs a reference to the ViewContainerRef where it can be added. You can provide a default ViewContainerRef for every Snackbar or provide a ViewContainerRef during each show call. A ViewContainerRef can be injected to every component in the constructor.

 });
Shows an auto hiding text.
showSnackbar(
snackbarMessage: IMdlSnackbarMessage)
: Promise<MdlSnackbarComponent>		 Shows a snackbar. The IMdlSnackbarMessage is: 
  {message: string;
  timeout?: number;
  action?:{
    handler: () => void;
    text: string;}
  vcRef?: ViewContainerRef;}

         
message
The message that should be shown. 
timeout
for how long should the message be visible. Defaults to 2750 ms. Is ignored if an action is proivded. 
action
Consists of a handler fuction and an aciton text. The text will be the buttons label. The action is called if the button is clicked. 
vcRef
The ViewContainerRef where the Snackbar is attached.

Returns a Promis with the created MdlSnackbarComponent. You can use this instance to manipulate the MdlSnackbarComponent and be informed if the Snackbar is shown to the user.
